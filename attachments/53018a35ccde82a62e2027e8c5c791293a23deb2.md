  228 |   private async resolveFieldValueLocator(label: string, initialValue: Locator, expected: string) {
  229 |     // Return the initial locator if it already contains text
  230 |     const initialText = (await initialValue.allTextContents())
  231 |       .join(" ")
  232 |       .replace(/\s+/g, " ")
  233 |       .trim();
  234 |     if (initialText) {
  235 |       return initialValue;
  236 |     }
  237 | 
  238 |     const labelLocator = this.page.getByText(label, { exact: true }).first();
  239 |     if (await labelLocator.count()) {
  240 |       // Locate the nearest container that holds both the label and its value
  241 |       const container = labelLocator.locator(
  242 |         "xpath=ancestor::*[self::div or self::li or self::tr or self::section or self::article][1]"
  243 |       );
  244 |       // Find non-label elements that may contain the field value
  245 |       const fallbackValue = container.locator(
  246 |         "xpath=.//*[not(self::label) and not(self::span[contains(@class, 'test-id__field-label')]) and normalize-space(.)!='']"
  247 |       );
  248 |       // Exclude elements whose text is only the label
  249 |       const candidate = fallbackValue.filter({ hasNotText: new RegExp(`^${label}$`, "i") }).first();
  250 |       if (await candidate.count()) {
  251 |         return candidate;
  252 |       }
  253 |       // Use the first available value element if no better match exists
  254 |       if (await fallbackValue.count()) {
  255 |         return fallbackValue.first();
  256 |       }
  257 |     }
  258 | 
  259 |     return this.page.locator("body");
  260 |   }
